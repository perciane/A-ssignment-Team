1. Task

Write a function for the knn regression estimator.

knn.reg <- function(t, x, y, k = 50) {
  d_t_x <- as.matrix(dist(c(t, x)))[1, -1]
  d_t_x_k <- sort(d_t_x, partial = k)[k]
  N_t_k <-  which(d_t_x <= d_t_x_k)
  m_hat <- sum(y[N_t_k]) / k
  return(m_hat)
}
